(X_train, y_train), (X_test, y_test) = cifar10.load_data()

# Normalize the images to range [0, 1]
X_train, X_test = X_train / 255.0, X_test / 255.0

# Convert labels to one-hot encoding
y_train = to_categorical(y_train, 10)
y_test = to_categorical(y_test, 10)

# Split training data into a smaller subset for object detection model (e.g., 10% of the data)
X_train_subset, _, y_train_subset, _ = train_test_split(X_train, y_train, test_size=0.9, random_state=42)

# Create ImageDataGenerator with augmentations
datagen = ImageDataGenerator(
    rotation_range=20,      # Random rotations
    width_shift_range=0.2,  # Random horizontal shifts
    height_shift_range=0.2, # Random vertical shifts
    shear_range=0.2,        # Shearing
    zoom_range=0.2,         # Zooming
    horizontal_flip=True,   # Horizontal flips

    fill_mode='nearest'     # Fill missing pixels
)

from sklearn.utils import class_weight

# Find unique classes in y_train_subset
unique_classes = np.unique(y_train_subset.argmax(axis=1))

# Compute class weights based on the unique classes present
class_weights = class_weight.compute_class_weight('balanced', classes=unique_classes, y=y_train_subset.argmax(axis=1))

# Convert class weights to a dictionary format
class_weight_dict = {unique_classes[i]: class_weights[i] for i in range(len(unique_classes))}

# Optionally print class weights to check
print(class_weight_dict)

# EfficientNetB0 model as a feature extractor (no top layer)
base_model = EfficientNetB0(weights='imagenet', include_top=False, input_shape=(32, 32, 3))

# Freeze base model layers
base_model.trainable = False

# Create the full model
model = models.Sequential([
    base_model,
    layers.GlobalAveragePooling2D(),
    layers.Dense(128, activation='relu'),
    layers.Dense(10, activation='softmax')  # 10 classes in CIFAR-10
])

model.compile(optimizer='adam', loss='categorical_crossentropy', metrics=['accuracy'])
